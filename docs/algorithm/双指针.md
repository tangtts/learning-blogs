function maxArea(height: number[]): number {
    let i = 0, j = height.length - 1, max = 0;
    while (i <= j) {
        // å³è¾¹æ¿å­æ›´çŸ­
        if (height[i] > height[j]) {
            max = Math.max((j - i) * height[j], max)
            j--;
        } else {
            max = Math.max((j - i) * height[i], max)
            i++;
        }
    }
    return max
};
